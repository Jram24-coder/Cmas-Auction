<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Christmas Auction Game</title>

    <style>

        body { font-family: Arial, sans-serif; text-align: center; padding: 20px; }

        .item { margin: 20px 0; }

        .bid-log { text-align: left; margin: 20px auto; width: 80%; }

        button { padding: 10px 20px; margin: 10px; cursor: pointer; }

    </style>

</head>

<body>

    <h1>Christmas Auction Game</h1>

    <h3 id="current-item">Item: A-1</h3>

    <p id="item-description">The ultimate tool for creating unforgettable moments...</p>

    <h2 id="highest-bid">Highest Bid: $0</h2>

    <div>

        <input id="bid-amount" type="number" placeholder="Enter your bid" />

        <input id="player-name" type="text" placeholder="Your Name" />

        <button onclick="placeBid()">Place Bid</button>

    </div>

    <div class="bid-log">

        <h3>Bid Log</h3>

        <ul id="log"></ul>

    </div>

    <button onclick="nextItem()">Next Item</button>

    <script>

        let currentItem = 1;

        const items = [

            { code: "A-1", description: "The ultimate tool for creating unforgettable moments..." },

            { code: "A-2", description: "A game-changer for anyone who values comfort and convenience." },

            { code: "A-3", description: "Transform your environment into something out of a dream." }

        ];

        let highestBid = 0;

        let bidLog = [];


        function placeBid() {

            const bidAmount = parseInt(document.getElementById("bid-amount").value);

            const playerName = document.getElementById("player-name").value;

            if (bidAmount > highestBid) {

                highestBid = bidAmount;

                document.getElementById("highest-bid").innerText = `Highest Bid: $${highestBid} by ${playerName}`;

                bidLog.push({ amount: bidAmount, name: playerName });

                updateLog();

            } else {

                alert("Bid must be higher than the current highest bid!");

            }

        }


        function updateLog() {

            const logElement = document.getElementById("log");

            logElement.innerHTML = "";

            bidLog.forEach((bid) => {

                const li = document.createElement("li");

                li.textContent = `${bid.name} bid $${bid.amount}`;

                logElement.appendChild(li);

            });

        }


        function nextItem() {

            currentItem++;

            if (currentItem > items.length) {

                alert("Auction is over!");

                return;

            }

            highestBid = 0;

            bidLog = [];

            document.getElementById("current-item").innerText = `Item: ${items[currentItem - 1].code}`;

            document.getElementById("item-description").innerText = items[currentItem - 1].description;

            document.getElementById("highest-bid").innerText = "Highest Bid: $0";

            updateLog();

        }

    </script>

</body>

</html>
